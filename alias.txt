alias cwd="pwd | xclip -i -sel p -f | xclip -i -sel c"
alias rxc="r | xclip -i -sel p -f | xclip -i -sel c"
alias tl="r|less"
function catl() {
  command cat $1 | less 
}
function listeningport() {
  if [[ -d $1 ]] ; then
    command netstat -tulpn
  else
    command netstat -tulpn | grep "$1"
  fi
}
